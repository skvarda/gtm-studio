function drawText(ctx, text, x, y, color, scale) {
  if (scale === undefined) scale = 1;
  ctx.fillStyle = color; let cx = x;
  for (let i = 0; i < text.length; i++) {
    const ch = text[i], glyph = FONT_DATA[ch];
    if (ch === ' ') { cx += 4*scale; continue; }
    if (!glyph) { cx += (FONT_W+1)*scale; continue; }
    for (let r = 0; r < FONT_H; r++) for (let c = 0; c < FONT_W; c++)
      if (glyph[r*FONT_W+c]) ctx.fillRect(cx+c*scale, y+r*scale, scale, scale);
    cx += (FONT_W+1)*scale;
  }
}
function measureText(text, scale) {
  if (scale === undefined) scale = 1; let w = 0;
  for (let i = 0; i < text.length; i++) w += (text[i]===' ') ? 4*scale : (FONT_W+1)*scale;
  if (text.length > 0) w -= 1*scale;
  return { width: w, height: FONT_H*scale };
}

function calculateBurnRate() {

function economyTick() {
  const churned = Math.floor(economy.customers * economy.churnRate);
  economy.customers = Math.max(0, economy.customers - churned);
  if (economy.salesPipeline > 0) {
    const c = Math.max(1, Math.floor(economy.salesPipeline * 0.3));
    economy.customers += c; economy.salesPipeline = Math.max(0, economy.salesPipeline - c);
  }
  if (economy.marketingLeads > 0) {
    const q = Math.max(1, Math.floor(economy.marketingLeads * 0.4));
    economy.salesPipeline += q; economy.marketingLeads = Math.max(0, economy.marketingLeads - q);
  }
  gameState.mrr = calculateMRR();
  economy.burnRate = calculateBurnRate();
  const net = gameState.mrr - economy.burnRate;
  gameState.cash += net;
  for (const e of economy.employees) if (e.type==='dev') economy.productQuality += e.skill * 0.5;
  return { revenue: gameState.mrr, burn: economy.burnRate, net, churned, newCustomers: economy.customers };
}
function getRunway() {
  const net = gameState.mrr - calculateBurnRate();
  if (net >= 0) return 999;
  return Math.floor(gameState.cash / Math.abs(net));
}

function isCollision(tx,ty) {
  if(tx<0||tx>=GRID_W||ty<2||ty>=GRID_H) return true;
  var room=ROOMS[gameState.currentRoom];
  if(room.collisions.has(ty+','+tx)) return true;
  if(room.npcs) for(var i=0;i<room.npcs.length;i++){var n=room.npcs[i]; if(n.x===tx&&n.y===ty) return true;}
  return false;
}

function formatCash(n) {
  if(n>=1000000) return '$'+(n/1000000).toFixed(1)+'M';
  if(n>=1000) return '$'+Math.floor(n/1000)+'K';
  return '$'+n;
}

const keysDown = {};
function handleInput(e) {
  keysDown[e.code]=true;
  if(['ArrowUp','ArrowDown','ArrowLeft','ArrowRight','Space','Tab'].includes(e.key)||
     ['ArrowUp','ArrowDown','ArrowLeft','ArrowRight','Space','Tab'].includes(e.code)) e.preventDefault();
  if(gameState.transition.active||gameState.locTransition.active) return;
  switch(gameState.screen) {
    case 'title': handleTitleInput(e.code); break;
    case 'nameEntry': handleNameEntryInput(e); break;
    case 'play': handlePlayInput(e.code); break;

// --- Movement ---
var MOVE_SPEED = 150;
function handlePlayInput(key) {
  if(key==='KeyE'||key==='Space'||key==='Enter') checkInteraction();
  else if(key==='KeyM'||key==='Tab') {
    if (gameState.day === 1 && !gameState.flags.roleChosen) return; // Block activities before role chosen
    showActivityMenu();
  }
}

function updateMovement(dt) {
  if(gameState.screen!=='play') return;
  var p=gameState.player;
  if(p.moving){
    p.moveProgress+=dt;
    var t=Math.min(p.moveProgress/MOVE_SPEED,1);
    var ease=t<0.5?2*t*t:1-Math.pow(-2*t+2,2)/2;
    var tdx=p.targetTileX*TILE, tdy=p.targetTileY*TILE+HUD_H;
    p.drawX=p.startDrawX+(tdx-p.startDrawX)*ease; p.drawY=p.startDrawY+(tdy-p.startDrawY)*ease;
    if(t>=1){p.tileX=p.targetTileX;p.tileY=p.targetTileY;p.drawX=tdx;p.drawY=tdy;p.moving=false;p.moveProgress=0;checkDoorTransition();}
    return;
  }
  var dx=0,dy=0,nf=null;
  if(keysDown['ArrowUp']||keysDown['KeyW']){dy=-1;nf='up';}
  else if(keysDown['ArrowDown']||keysDown['KeyS']){dy=1;nf='down';}
  else if(keysDown['ArrowLeft']||keysDown['KeyA']){dx=-1;nf='left';}
  else if(keysDown['ArrowRight']||keysDown['KeyD']){dx=1;nf='right';}
  if(nf){
    p.facing=nf; var nx=p.tileX+dx,ny=p.tileY+dy;
    if(!isCollision(nx,ny)){p.targetTileX=nx;p.targetTileY=ny;p.startDrawX=p.drawX;p.startDrawY=p.drawY;p.moving=true;p.moveProgress=0;}
  }
}

function checkDoorTransition() {
  var room=ROOMS[gameState.currentRoom],p=gameState.player;
  if(!room.doors) return;
  for(var i=0;i<room.doors.length;i++){var door=room.doors[i]; if(p.tileX===door.x&&p.tileY===door.y) { transitionToRoom(door.target,door.spawn); return; }}
}

// --- Interaction ---
function getFacingTile() {
  var p=gameState.player; var tx=p.tileX,ty=p.tileY;
  if(p.facing==='up')ty--;else if(p.facing==='down')ty++;else if(p.facing==='left')tx--;else if(p.facing==='right')tx++;
  return {x:tx,y:ty};
}

function checkInteraction() {
  var ft=getFacingTile(), room=ROOMS[gameState.currentRoom];
  // NPC interaction
  if(room.npcs) for(var i=0;i<room.npcs.length;i++){var npc=room.npcs[i]; if(ft.x===npc.x&&ft.y===npc.y){
    var result = getNPCDialogue(npc.id);
    if (result.afterDialogue === 'roleChoice') {
      // Show Jordan's intro, then present role choice
      showDialogueWithCallback(NPC_DATA[npc.id].name, result.lines, function() {
        showChoice([
          {id:'seller', label:'THE SELLER - Sales focus'},
          {id:'dev', label:'THE DEV - Technical focus'},
          {id:'marketer', label:'THE MARKETER - Growth focus'}
        ], function(option) {
          var roleData = getRoleData(option.id);
          gameState.player.role = option.id;
          gameState.player.stats = Object.assign({}, roleData.stats);
          gameState.flags.roleChosen = true;
          // Show Jordan's response to role choice
          var responseLines = STORY_DIALOGUES.jordan.day_1_post_role[option.id];
          showDialogue('Jordan', responseLines);
        });
      });
      return;
    }
    showDialogue(NPC_DATA[npc.id].name, result.lines);
    return;
  }}
  // Interactable interaction
  if(room.interactables) for(var j=0;j<room.interactables.length;j++){var obj=room.interactables[j]; if(ft.x===obj.x&&ft.y===obj.y){
    if(obj.id==='laptop') {
      if (gameState.day === 1 && !gameState.flags.laptopDay1Seen) {
        gameState.flags.laptopDay1Seen = true;
        showDialogueWithCallback('Laptop', STORY_DIALOGUES.laptop.day_1_first, function() {
          showLaptopMenu();
        });
      } else {
        showLaptopMenu();
      }
    }
    else if(obj.id==='whiteboard') {
      if (gameState.day === 1 && !gameState.flags.whiteboardDay1Seen) {
        gameState.flags.whiteboardDay1Seen = true;
        showDialogue('Whiteboard', STORY_DIALOGUES.whiteboard.day_1);

        showDialogue('Whiteboard',['Quality: '+Math.floor(economy.productQuality),'Pipeline: '+economy.salesPipeline,'Leads: '+economy.marketingLeads,'Team: '+economy.employees.length]);
      }
    }
    return;
  }}
  // Door interaction
  if(room.doors) for(var k=0;k<room.doors.length;k++){var door=room.doors[k]; if(ft.x===door.x&&ft.y===door.y){ transitionToRoom(door.target,door.spawn); return; }}
}

function advanceDialogue() {

    if (cb) { cb(); }

  }

function wrapText(text,maxWidth,scale) {
  var words=text.split(' '),lines=[]; var cur='';
  for(var i=0;i<words.length;i++){var w=words[i];var test=cur?cur+' '+w:w;if(measureText(test,scale).width>maxWidth){if(cur)lines.push(cur);cur=w;}else cur=test;}
  if(cur) lines.push(cur); return lines;
}

function resetAP() { gameState.ap=gameState.maxAP; }
function advanceDay() {
  var report=economyTick(); gameState.dayReport=report;
  gameState.screen='dayTransition'; gameState.dayTransition={timer:0,phase:'fadeOut'};
}
function updateDayTransition(dt) {
  var t=gameState.dayTransition; t.timer+=dt;
  if(t.phase==='fadeOut'&&t.timer>=400){gameState.day++;resetAP();t.phase='splash';t.timer=0;}
  else if(t.phase==='splash'&&t.timer>=1200){t.phase='fadeIn';t.timer=0;}
  else if(t.phase==='fadeIn'&&t.timer>=300) gameState.screen='dayReport';
}
function drawDayTransition(ctx) {
  var t=gameState.dayTransition;
  if(t.phase==='fadeOut'){drawPlayScreen(ctx);ctx.fillStyle='rgba(0,0,0,'+Math.min(t.timer/400,1)+')';ctx.fillRect(0,0,640,360);}
  else if(t.phase==='splash'){
    ctx.fillStyle=PAL.black; ctx.fillRect(0,0,640,360);
    var ds='DAY '+gameState.day; drawText(ctx,ds,Math.floor((640-measureText(ds,4).width)/2),120,PAL.gold,4);
    var ws='WEEK '+gameState.day; drawText(ctx,ws,Math.floor((640-measureText(ws,2).width)/2),170,PAL.teal,2);
    drawText(ctx,formatCash(gameState.cash),Math.floor((640-measureText(formatCash(gameState.cash),2).width)/2),210,PAL.gold,2);
    if(gameState.cash<=0){var go='GAME OVER';drawText(ctx,go,Math.floor((640-measureText(go,3).width)/2),260,PAL.danger,3);}
  } else if(t.phase==='fadeIn'){drawPlayScreen(ctx);ctx.fillStyle='rgba(0,0,0,'+(1-Math.min(t.timer/300,1))+')';ctx.fillRect(0,0,640,360);}
}

// --- Game Loop ---
var lastTime = 0;
function gameLoop(timestamp) {
  if(!lastTime) lastTime=timestamp;
  var dt=timestamp-lastTime; lastTime=timestamp;
  if(dt>50) dt=50;
  update(dt); draw(ctx);
  requestAnimationFrame(gameLoop);
}
function update(dt) {
  updateTransition(dt); updateLocTransition(dt);
  if(gameState.screen==='play') updateMovement(dt);
  else if(gameState.screen==='dayTransition') updateDayTransition(dt);
}
function draw(ctx) {
  ctx.fillStyle=PAL.bg; ctx.fillRect(0,0,640,360);
  switch(gameState.screen) {
    case 'title': drawTitleScreen(ctx); break;
    case 'nameEntry': drawNameEntry(ctx); break;
    case 'play': drawPlayScreen(ctx); break;
    case 'dialogue': drawPlayScreen(ctx); drawDialogue(ctx); break;
    case 'choice': drawChoiceBox(ctx); break;
    case 'laptopMenu': drawPlayScreen(ctx); drawLaptopMenu(ctx); break;
    case 'activityMenu': drawPlayScreen(ctx); drawActivityMenu(ctx); break;
    case 'emailInbox': drawPlayScreen(ctx); drawEmailInbox(ctx); break;
    case 'dayTransition': drawDayTransition(ctx); break;
    case 'dayReport': drawPlayScreen(ctx); drawDayReport(ctx); break;
  }
  drawTransitionOverlay(ctx); drawLocTransitionOverlay(ctx);
}
