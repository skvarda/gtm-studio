function drawText(ctx,text,x,y,color,scale){
  if(scale===undefined)scale=1;ctx.fillStyle=color;var cx=x;
  for(var i=0;i<text.length;i++){var ch=text[i],glyph=FONT_DATA[ch];
    if(ch===' '){cx+=4*scale;continue;}if(!glyph){cx+=(FONT_W+1)*scale;continue;}
    for(var r=0;r<FONT_H;r++)for(var c=0;c<FONT_W;c++)if(glyph[r*FONT_W+c])ctx.fillRect(cx+c*scale,y+r*scale,scale,scale);

}
function measureText(text,scale){
  if(scale===undefined)scale=1;var w=0;
  for(var i=0;i<text.length;i++)w+=(text[i]===' ')?4*scale:(FONT_W+1)*scale;
  if(text.length>0)w-=1*scale;return{width:w,height:FONT_H*scale};
}
function wrapText(text,maxWidth,scale){
  var words=text.split(' '),lines=[],cur='';
  for(var i=0;i<words.length;i++){var w=words[i],test=cur?cur+' '+w:w;
    if(measureText(test,scale).width>maxWidth){if(cur)lines.push(cur);cur=w;}else cur=test;}
  if(cur)lines.push(cur);return lines;
}

function tryMove(dir){
  if(player.moving)return;player.dir=dir;
  var nx=player.gx+DX[dir],ny=player.gy+DY[dir];
  if(!isTileWalkable(nx,ny))return;if(getNPCAt(nx,ny))return;
  player.gx=nx;player.gy=ny;player.sx=player.px;player.sy=player.py;
  player.tx=nx*TILE;player.ty=ny*TILE+MAP_Y;player.moving=true;player.mt=0;
}

function updatePlayer(dt){
  if(!player.moving)return;player.mt+=dt;
  var t=Math.min(1,player.mt/MOVE_MS);t=1-(1-t)*(1-t);
  player.px=player.sx+(player.tx-player.sx)*t;player.py=player.sy+(player.ty-player.sy)*t;
  if(player.mt>=MOVE_MS){player.px=player.tx;player.py=player.ty;player.moving=false;}
}

// Get the next unseen story dialogue for a source on the current day
function getStoryDialogue(source){
  var d = gameState.day;
  var key = 'day_' + d;
  var pool = STORY_DIALOGUES[source] && STORY_DIALOGUES[source][key];
  if(!pool || pool.length === 0) return null;
  // Find first unseen dialogue
  for(var i = 0; i < pool.length; i++){
    if(!gameState.flags['seen_' + pool[i].id]){
      return pool[i];
    }
  }
  // All seen for today, return last one as repeatable
  return pool[pool.length - 1];
}

function markStoryDialogueSeen(id){
  gameState.flags['seen_' + id] = true;
}

function startDialogue(speaker, lines, color){
  dlg.active=true; dlg.speaker=speaker; dlg.lines=lines; dlg.cur=0;
  dlg.color=color||PAL.textPrimary; dlg.choices=null; dlg.choiceIndex=0;
  dlg.storyId=null; dlg.onChoice=null;

function startStoryDialogue(sd){
  dlg.active=true; dlg.speaker=sd.speaker; dlg.lines=[sd.text]; dlg.cur=0;
  dlg.color=PAL.textAccent; dlg.choices=sd.choices||null; dlg.choiceIndex=0;
  dlg.storyId=sd.id; dlg.onChoice=function(ci){
    if(sd.choices && sd.choices[ci] && sd.choices[ci].effect){
      var ef = sd.choices[ci].effect;
      if(ef.jordan_relationship){
        if(!gameState.flags.jordan_relationship) gameState.flags.jordan_relationship = 0;
        gameState.flags.jordan_relationship += ef.jordan_relationship;
      }
    }
    markStoryDialogueSeen(sd.id);
  };
  gameState.uiState='dialogue';
}

function advanceDialogue(){
  // If we're showing choices, don't advance with space — need selection
  if(dlg.choices && dlg.cur >= dlg.lines.length - 1){
    return; // must use choice selection
  }
  dlg.cur++;
  if(dlg.cur>=dlg.lines.length){
    if(dlg.choices){
      // Show choices after last line — already showing
      return;
    }
    dlg.active=false; gameState.uiState='play';
    if(dlg.storyId) markStoryDialogueSeen(dlg.storyId);
  }
}

function selectDialogueChoice(ci){
  if(dlg.onChoice) dlg.onChoice(ci);
  dlg.active=false; dlg.choices=null; gameState.uiState='play';
}

function drawDialogueBox(ctx){
  if(!dlg.active) return;
  var bh, by;
  var atEnd = dlg.cur >= dlg.lines.length - 1;
  var showingChoices = atEnd && dlg.choices && dlg.choices.length > 0;

  if(showingChoices){
    bh = 80 + dlg.choices.length * 20 + 8;
    by = 360 - bh;
  } else {
    bh = 80; by = 280;
  }

  ctx.fillStyle=PAL.emailBg; ctx.fillRect(0,by,640,bh);
  ctx.fillStyle=PAL.uiBorder; ctx.fillRect(0,by,640,2);
  drawText(ctx,dlg.speaker,16,by+8,dlg.color,2);

  if(dlg.cur<dlg.lines.length){
    var tl=wrapText(dlg.lines[dlg.cur],600,1);
    for(var i=0;i<tl.length;i++) drawText(ctx,tl[i],16,by+32+i*14,PAL.textPrimary,1);

    if(showingChoices){
      var cy = by + 32 + tl.length * 14 + 8;
      for(var c=0; c<dlg.choices.length; c++){
        var isA = c === dlg.choiceIndex;
        if(isA){
          ctx.fillStyle=PAL.choiceActive; ctx.fillRect(14,cy-2,612,18);
          drawText(ctx,'>',16,cy,PAL.textAccent,1);
        }
        drawText(ctx, dlg.choices[c].text, 28, cy, isA?PAL.textPrimary:PAL.textSecondary, 1);
        cy += 20;
      }
    } else {
      if(Math.floor(Date.now()/400)%2===0) drawText(ctx,'SPACE',580,by+bh-16,PAL.textDim,1);
    }
  }
}

function performAction(actionId){
  var action=null;for(var i=0;i<ENGINE_ACTIONS.length;i++)if(ENGINE_ACTIONS[i].id===actionId){action=ENGINE_ACTIONS[i];break;}
  if(!action)return{success:false,message:'Unknown action.'};
  var cost=AP.costs[action.apKey]||0;
  if(cost>0&&AP.current<cost)return{success:false,message:'Not enough AP.'};
  if(!action.enabled(gameState))return{success:false,message:'Action not available.'};
  if(cost>0)spendAP(action.apKey);
  var result=action.execute(gameState);
  return{success:true,message:result.text,effects:result.effects};
}

function economyTick(){
  var gs=gameState;var wRev=gs.customers*23;var empSal=0;
  for(var i=0;i<gs.employees.length;i++)empSal+=Math.floor((gs.employees[i].salary||0)/4);
  var wBurn=2000+empSal;gs.cash+=wRev;gs.cash-=wBurn;gs.mrr=gs.customers*99;
  var churned=0;
  if(gs.customers>5){var cr=gs.bugs>10?0.06:0.03;for(var i=0;i<gs.customers;i++)if(Math.random()<cr)churned++;churned=Math.min(churned,gs.customers);gs.customers=Math.max(0,gs.customers-churned);gs.mrr=gs.customers*99;}
  if(gs.product>0&&Math.random()<0.25)gs.bugs+=1;
  var nb=wBurn-wRev;var rw;if(nb<=0)rw=999;else if(gs.cash<=0)rw=0;else rw=Math.floor(gs.cash/nb);
  if(gs.cash<=0&&rw<=0)gs.gameOver=true;
  return{revenue:wRev,burn:wBurn,churnedCustomers:churned,runway:rw,cashChange:wRev-wBurn};
}

function generateDailyEmails(day){
  var gs=gameState, emails=[];
  var dayKey = 'day_' + day;

  // Check for story emails first (new curated content)
  if(STORY_EMAILS[dayKey]){
    var storyEmails = STORY_EMAILS[dayKey];
    for(var i = 0; i < storyEmails.length; i++){
      if(gs.emailsRead.indexOf(storyEmails[i].id) === -1){
        var se = storyEmails[i];
        gs.emailsRead.push(se.id);
        emails.push({id:se.id, from:se.from, subject:se.subject, body:se.body, read:false, choices:se.choices ? se.choices.slice() : null});
      }
    }
  }

  // Then add template-based emails
  var mandatory=[],storyPool=[],regularPool=[];
  for(var i=0;i<EMAIL_TEMPLATES.length;i++){var t=EMAIL_TEMPLATES[i];
    if(day<t.day_range[0]||day>t.day_range[1])continue;if(t.unique&&gs.emailsRead.indexOf(t.id)!==-1)continue;
    if(t.day_range[0]===t.day_range[1]&&t.day_range[0]===day)mandatory.push(t);
    else if(t.unique&&t.category==='story')storyPool.push(t);else regularPool.push(t);
  }
  for(var i=0;i<mandatory.length;i++)emails.push(makeEmail(mandatory[i]));
  storyPool.sort(function(){return Math.random()-0.5;});
  for(var i=0;i<storyPool.length&&emails.length<3;i++)emails.push(makeEmail(storyPool[i]));
  var target=3+Math.floor(Math.random()*2);
  regularPool.sort(function(){return Math.random()-0.5;});
  for(var i=0;i<regularPool.length&&emails.length<target;i++)emails.push(makeEmail(regularPool[i]));
  if(emails.length===0)emails.push({id:'fallback_'+day,from:'Jordan',subject:'Day '+day,body:'Another day. Get to work.',read:false,choices:null});
  return emails;
}
function makeEmail(t){if(t.unique)gameState.emailsRead.push(t.id);return{id:t.id,from:t.from,subject:t.subject,body:t.body,read:false,choices:t.choices?t.choices.slice():null};}
function initEmailPhase(){var gs=gameState;gs.phase='morning';gs.inbox=generateDailyEmails(gs.day);gs.uiState='inbox';gs.selectedIndex=0;gs.choiceIndex=0;gs.actionResult=null;gs.eveningSummary=null;}
function exitEmailPhase(){startAfternoon();}
function markEmailRead(i){if(i>=0&&i<gameState.inbox.length)gameState.inbox[i].read=true;}
function selectEmailChoice(email,ci){
  if(!email.choices||ci<0||ci>=email.choices.length)return;var ef=email.choices[ci].effect;if(!ef)return;var gs=gameState;
  if(ef.flag)gs.flags[ef.flag]=true;if(ef.morale)gs.morale+=ef.morale;if(ef.customers)gs.customers+=ef.customers;
  if(ef.product)gs.product+=ef.product;if(ef.cash)gs.cash+=ef.cash;
}

function handleInput(e){
  var key=e.code;
  if(['ArrowUp','ArrowDown','ArrowLeft','ArrowRight','Space','Tab','Enter','KeyE','KeyM','KeyW','KeyA','KeyS','KeyD'].indexOf(key)!==-1)e.preventDefault();
  var gs=gameState;

  if(gs.uiState==='title'){if(key==='Enter'||key==='Space'){gs.uiState='character_select';gs.selectedIndex=1;}return;}
  if(gs.uiState==='character_select'){
    if(key==='ArrowLeft'||key==='KeyA')gs.selectedIndex=Math.max(0,gs.selectedIndex-1);
    else if(key==='ArrowRight'||key==='KeyD')gs.selectedIndex=Math.min(ROLES.length-1,gs.selectedIndex+1);
    else if(key==='Enter'||key==='Space'){
      gs.role=ROLES[gs.selectedIndex].id;ROLES[gs.selectedIndex].apply(gs);
      startDay(1);
    }return;
  }
  if(gs.gameOver&&gs.uiState==='game_over'){if(key==='Enter'||key==='Space')location.reload();return;}

  switch(gs.uiState){
    case 'inbox':handleInboxInput(key);break;
    case 'email_choice':handleEmailChoiceInput(key);break;
    case 'play':handlePlayInput(key);break;
    case 'dialogue':handleDialogueInput(key);break;
    case 'action_menu':handleActionMenuInput(key);break;
    case 'action_result':handleActionResultInput(key);break;
    case 'evening_summary':handleEveningSummaryInput(key);break;
  }
}

function handleInboxInput(key){
  var gs=gameState,inbox=gs.inbox;
  if(key==='ArrowUp'||key==='KeyW')gs.selectedIndex=Math.max(0,gs.selectedIndex-1);
  else if(key==='ArrowDown'||key==='KeyS')gs.selectedIndex=Math.min(inbox.length-1,gs.selectedIndex+1);
  else if(key==='Enter'||key==='Space'){
    if(gs.selectedIndex>=0&&gs.selectedIndex<inbox.length){var em=inbox[gs.selectedIndex];
      if(!em.read){if(em.choices&&em.choices.length>0){gs.uiState='email_choice';gs.choiceIndex=0;}else markEmailRead(gs.selectedIndex);}
    }
  }else if(key==='KeyE'){if(allEmailsRead())exitEmailPhase();}
}

function handleEmailChoiceInput(key){
  var gs=gameState,em=gs.inbox[gs.selectedIndex];if(!em||!em.choices){gs.uiState='inbox';return;}
  if(key==='ArrowUp'||key==='KeyW')gs.choiceIndex=Math.max(0,gs.choiceIndex-1);
  else if(key==='ArrowDown'||key==='KeyS')gs.choiceIndex=Math.min(em.choices.length-1,gs.choiceIndex+1);
  else if(key==='Enter'||key==='Space'){selectEmailChoice(em,gs.choiceIndex);markEmailRead(gs.selectedIndex);gs.uiState='inbox';}
  else if(key==='Escape'||key==='KeyX')gs.uiState='inbox';
}

function handlePlayInput(key){
  var gs=gameState;
  if(key==='KeyM'||key==='Tab'){gs.uiState='action_menu';gs.selectedIndex=0;return;}
  if(key==='KeyE'||key==='Space'||key==='Enter'){
    if(player.moving)return;
    var fx=player.gx+DX[player.dir],fy=player.gy+DY[player.dir];
    var npc=getNPCAt(fx,fy);
    if(npc){
      if(npc.id==='jordan'){
        // Try story dialogue first
        var sd = getStoryDialogue('jordan');
        if(sd && !gameState.flags['seen_' + sd.id]){
          startStoryDialogue(sd);
          return;
        }
        // Fallback to generic lines
        var lines = getJordanLines();
        if(lines) startDialogue(npc.name, lines, PAL.textAccent);
      }
      else if(npc.id==='maya'){startDialogue(npc.name,['Working on the new feature!','Jordan\'s architecture is wild.','But I\'m getting the hang of it.'],PAL.textAccent);}
      else if(npc.id==='derek'){startDialogue(npc.name,['LET\'S CLOSE SOME DEALS!','*polishes sales gong*','I smell a commission.'],PAL.textAccent);}
      else startDialogue(npc.name,['...'],PAL.textAccent);
      return;
    }
    var tile=getTileAt(fx,fy);var actKey=TILE_ACTIONS[tile];
    if(actKey==='door'){gs.uiState='action_menu';gs.selectedIndex=0;return;}

    // Story interactions for laptop and whiteboard
    if(actKey==='computer'){
      var sd = getStoryDialogue('laptop');
      if(sd && !gameState.flags['seen_' + sd.id]){
        startStoryDialogue(sd);
        return;
      }
      // Fallback: code action from tile
      var result = performAction('code');
      if(result.success){gs.actionResult=result;gs.uiState='action_result';}
      else showFlash(result.message);
      return;
    }
    if(actKey==='whiteboard'){
      var sd = getStoryDialogue('whiteboard');
      if(sd && !gameState.flags['seen_' + sd.id]){
        startStoryDialogue(sd);
        return;
      }
      // Fallback: plan action from tile
      var result = performAction('plan');
      if(result.success){gs.actionResult=result;gs.uiState='action_result';}
      else showFlash(result.message);
      return;
    }

    if(actKey){
      var result=performAction(actKey);
      if(result.success){gs.actionResult=result;gs.uiState='action_result';}
      else showFlash(result.message);
    }
    return;
  }
}

function handleDialogueInput(key){
  var atEnd = dlg.cur >= dlg.lines.length - 1;
  var showingChoices = atEnd && dlg.choices && dlg.choices.length > 0;

  if(showingChoices){
    if(key==='ArrowUp'||key==='KeyW') dlg.choiceIndex = Math.max(0, dlg.choiceIndex - 1);
    else if(key==='ArrowDown'||key==='KeyS') dlg.choiceIndex = Math.min(dlg.choices.length - 1, dlg.choiceIndex + 1);
    else if(key==='Enter'||key==='Space'||key==='KeyE') selectDialogueChoice(dlg.choiceIndex);
  } else {
    if(key==='Space'||key==='Enter'||key==='KeyE') advanceDialogue();
  }
}

function handleActionMenuInput(key){
  var gs=gameState,actions=ENGINE_ACTIONS;
  if(key==='Escape'||key==='KeyX'){gs.uiState='play';return;}
  if(key==='ArrowUp'||key==='KeyW')gs.selectedIndex=Math.max(0,gs.selectedIndex-1);
  else if(key==='ArrowDown'||key==='KeyS')gs.selectedIndex=Math.min(actions.length-1,gs.selectedIndex+1);
  else if(key==='Enter'||key==='Space'){
    var a=actions[gs.selectedIndex];if(!a)return;
    var cost=AP.costs[a.apKey]||0,aff=cost===0||AP.current>=cost,en=a.enabled(gs)&&aff;
    if(!en)return;var result=performAction(a.id);
    if(result.success){gs.actionResult=result;gs.uiState='action_result';}
  }
}

function handleActionResultInput(key){
  var gs=gameState;
  if(key==='Space'||key==='Enter'){
    gs.actionResult=null;
    if(AP.current<=0)startEvening();
    else gs.uiState='play';
  }
}

function handleEveningSummaryInput(key){
  if(key==='Enter'||key==='Space'){
    if(gameState.gameOver)gameState.uiState='game_over';
    else advanceDay();
  }
}

    if(!player.moving){
      if(keysDown['ArrowUp']||keysDown['KeyW'])tryMove(DIR_UP);

      else if(keysDown['ArrowLeft']||keysDown['KeyA'])tryMove(DIR_LT);
      else if(keysDown['ArrowRight']||keysDown['KeyD'])tryMove(DIR_RT);
    }
  }
  if(flashTimer>0)flashTimer-=dt;

  ctx.fillStyle=PAL.bg;ctx.fillRect(0,0,640,360);
  switch(gameState.uiState){
    case 'title':drawTitle(ctx);break;
    case 'character_select':drawCharSelect(ctx);break;

    case 'action_result':drawActionResult(ctx);break;
    case 'evening_summary':drawEveningSummary(ctx);break;
    case 'game_over':drawGameOver(ctx);break;
  }

  requestAnimationFrame(gameLoop);
}
