function drawText(ctx, text, x, y, color, scale) {
  if (scale === undefined) scale = 1;
  ctx.fillStyle = color;
  let cx = x;
  for (let i = 0; i < text.length; i++) {
    const ch = text[i], glyph = FONT_DATA[ch];
    if (ch === ' ') { cx += 4 * scale; continue; }
    if (!glyph) { cx += (FONT_W + 1) * scale; continue; }
    for (let r = 0; r < FONT_H; r++)
      for (let c = 0; c < FONT_W; c++)
        if (glyph[r * FONT_W + c]) ctx.fillRect(cx + c * scale, y + r * scale, scale, scale);
    cx += (FONT_W + 1) * scale;
  }
}

function measureText(text, scale) {
  if (scale === undefined) scale = 1;
  let w = 0;
  for (let i = 0; i < text.length; i++) w += (text[i] === ' ') ? 4 * scale : (FONT_W + 1) * scale;
  if (text.length > 0) w -= 1 * scale;
  return { width: w, height: FONT_H * scale };
}

function wrapText(text, maxWidth, scale) {
  const words = text.split(' '), lines = [];
  let cur = '';
  for (let i = 0; i < words.length; i++) {
    const w = words[i];
    const test = cur ? cur + ' ' + w : w;
    if (measureText(test, scale).width > maxWidth) {
      if (cur) lines.push(cur);
      cur = w;
    } else cur = test;
  }
  if (cur) lines.push(cur);
  return lines;
}

// --- Format helpers ---
function formatCash(amount) {
  if (amount < 0) return '-$' + formatCashAbs(-amount);
  return '$' + formatCashAbs(amount);
}
function formatCashAbs(n) {
  if (n >= 1000000) return (n / 1000000).toFixed(1) + 'M';
  if (n >= 10000) return Math.floor(n / 1000) + 'K';
  return Math.floor(n).toString().replace(/\B(?=(\d{3})+(?!\d))/g, ',');
}
function formatRunway(weeks) {
  if (weeks >= 999) return '~';
  if (weeks <= 0) return '0w';
  return weeks + 'w';
}

function performAction(actionId) {
  var action = null;
  for (var i = 0; i < ENGINE_ACTIONS.length; i++) {
    if (ENGINE_ACTIONS[i].id === actionId) { action = ENGINE_ACTIONS[i]; break; }
  }
  if (!action) return { success: false, message: 'Unknown action.' };
  var cost = AP.costs[action.apKey] || 0;
  if (cost > 0 && AP.current < cost) return { success: false, message: 'Not enough AP.' };
  if (!action.enabled(gameState)) return { success: false, message: 'Action not available.' };
  if (cost > 0) spendAP(action.apKey);
  var result = action.execute(gameState);
  return { success: true, message: result.text, effects: result.effects };
}

function economyTick() {
  var gs = gameState;
  var weeklyRevPerCustomer = 23;
  var weeklyRevenue = gs.customers * weeklyRevPerCustomer;
  var employeeSalaries = 0;
  for (var i = 0; i < gs.employees.length; i++) employeeSalaries += Math.floor((gs.employees[i].salary || 0) / 4);
  var weeklyBurn = 2000 + employeeSalaries;

  gs.cash += weeklyRevenue;
  gs.cash -= weeklyBurn;
  gs.mrr = gs.customers * 99;

  var churnedCustomers = 0;
  if (gs.customers > 5) {
    var churnRate = 0.03;
    if (gs.bugs > 10) churnRate = 0.06;
    for (var i = 0; i < gs.customers; i++) {
      if (Math.random() < churnRate) churnedCustomers++;
    }
    churnedCustomers = Math.min(churnedCustomers, gs.customers);
    gs.customers = Math.max(0, gs.customers - churnedCustomers);
    gs.mrr = gs.customers * 99;
  }

  if (gs.product > 0 && Math.random() < 0.25) gs.bugs += 1;

  var netBurn = weeklyBurn - weeklyRevenue;
  var runway;
  if (netBurn <= 0) runway = 999;
  else if (gs.cash <= 0) runway = 0;
  else runway = Math.floor(gs.cash / netBurn);

  if (gs.cash <= 0 && runway <= 0) gs.gameOver = true;

  return {
    revenue: weeklyRevenue,
    burn: weeklyBurn,
    churnedCustomers: churnedCustomers,
    runway: runway,
    cashChange: weeklyRevenue - weeklyBurn
  };
}

function generateDailyEmails(day) {
  var gs = gameState;
  var emails = [];
  var mandatory = [];
  var storyPool = [];
  var regularPool = [];

  for (var i = 0; i < EMAIL_TEMPLATES.length; i++) {
    var tmpl = EMAIL_TEMPLATES[i];
    if (day < tmpl.day_range[0] || day > tmpl.day_range[1]) continue;
    if (tmpl.unique && gs.emailsRead.indexOf(tmpl.id) !== -1) continue;

    if (tmpl.day_range[0] === tmpl.day_range[1] && tmpl.day_range[0] === day) {
      mandatory.push(tmpl);
    } else if (tmpl.unique && tmpl.category === 'story') {
      storyPool.push(tmpl);
    } else {
      regularPool.push(tmpl);
    }
  }

  for (var i = 0; i < mandatory.length; i++) {
    emails.push(makeEmail(mandatory[i]));
  }

  var shuffledStory = storyPool.sort(function() { return Math.random() - 0.5; });
  for (var i = 0; i < shuffledStory.length && emails.length < 2; i++) {
    emails.push(makeEmail(shuffledStory[i]));
  }

  var target = 2 + Math.floor(Math.random() * 3);
  var shuffledRegular = regularPool.sort(function() { return Math.random() - 0.5; });
  for (var i = 0; i < shuffledRegular.length && emails.length < target; i++) {
    emails.push(makeEmail(shuffledRegular[i]));
  }

  if (emails.length === 0) {
    emails.push({
      id: 'fallback_' + day,
      from: 'Jordan',
      subject: 'Day ' + day,
      body: 'Another day, another chance to not go bankrupt. Get to work.',
      read: false,
      choices: null
    });
  }

  return emails;
}

function makeEmail(tmpl) {
  if (tmpl.unique) gameState.emailsRead.push(tmpl.id);
  return {
    id: tmpl.id,
    from: tmpl.from,
    subject: tmpl.subject,
    body: tmpl.body,
    read: false,
    choices: tmpl.choices ? tmpl.choices.slice() : null
  };
}

function initEmailPhase() {
  var gs = gameState;
  gs.phase = 'morning';
  gs.inbox = generateDailyEmails(gs.day);
  gs.uiState = 'inbox';
  gs.selectedIndex = 0;
  gs.choiceIndex = 0;
  gs.actionResult = null;
  gs.eveningSummary = null;
}

function exitEmailPhase() {
  startAfternoon();
}

function markEmailRead(index) {
  if (index >= 0 && index < gameState.inbox.length) {
    gameState.inbox[index].read = true;
  }
}

function selectEmailChoice(email, choiceIndex) {
  if (!email.choices || choiceIndex < 0 || choiceIndex >= email.choices.length) return;
  var effect = email.choices[choiceIndex].effect;
  if (!effect) return;
  var gs = gameState;
  if (effect.flag) gs.flags[effect.flag] = true;
  if (effect.morale) gs.morale += effect.morale;
  if (effect.customers) gs.customers += effect.customers;
  if (effect.product) gs.product += effect.product;
  if (effect.cash) gs.cash += effect.cash;
  if (effect.ap_bonus) { AP.current = Math.min(AP.max, AP.current + effect.ap_bonus); gs.ap = AP.current; }
}

function allEmailsRead() {
  var inbox = gameState.inbox;
  for (var i = 0; i < inbox.length; i++) {
    if (!inbox[i].read) return false;
  }
  return true;
}

function drawHUD(ctx) {
  var gs = gameState;

  // Background bar
  ctx.fillStyle = PAL.hudBg;
  ctx.fillRect(0, 0, 640, 28);
  ctx.fillStyle = '#2a2a3e';
  ctx.fillRect(0, 28, 640, 1);

  // Day
  drawText(ctx, 'DAY ' + gs.day, 6, 4, PAL.hudText, 1);

  // Phase indicator with color
  var phaseColor = gs.phase === 'morning' ? PAL.morning : (gs.phase === 'afternoon' ? PAL.afternoon : PAL.evening);
  drawText(ctx, gs.phase.toUpperCase(), 6, 16, phaseColor, 1);

  // Cash
  var cashColor = gs.cash < 0 ? PAL.red : PAL.textAccent;
  drawText(ctx, formatCash(gs.cash), 100, 4, cashColor, 1);

  // MRR
  drawText(ctx, 'MRR ' + formatCash(gs.mrr), 220, 4, PAL.positive, 1);

  // Runway
  var weeklyBurn = 2000;
  for (var i = 0; i < gs.employees.length; i++) weeklyBurn += Math.floor((gs.employees[i].salary || 0) / 4);
  var weeklyRev = gs.customers * 23;
  var runwayWeeks;
  if (weeklyBurn <= weeklyRev) runwayWeeks = 999;
  else if (gs.cash <= 0) runwayWeeks = 0;
  else runwayWeeks = Math.floor(gs.cash / (weeklyBurn - weeklyRev));

  var rwStr = 'RW ' + formatRunway(runwayWeeks);
  var rwColor = runwayWeeks >= 999 ? PAL.positive : (runwayWeeks < 24 ? PAL.negative : PAL.hudText);
  drawText(ctx, rwStr, 360, 4, rwColor, 1);

  // AP pips
  drawText(ctx, 'AP', 470, 8, PAL.hudText, 1);
  drawAPPips(ctx, 499, 11);

  // Era
  drawText(ctx, gs.era.toUpperCase(), 590, 16, PAL.textDim, 1);
}

function drawInbox(ctx) {
  var gs = gameState;
  var inbox = gs.inbox;
  var sel = gs.selectedIndex;

  ctx.fillStyle = PAL.bg;
  ctx.fillRect(0, 0, 640, 360);

  // Header
  ctx.fillStyle = PAL.hudBg;
  ctx.fillRect(0, 0, 640, 30);
  drawText(ctx, 'INBOX - DAY ' + gs.day + ' MORNING', 10, 8, PAL.morning, 2);

  var unreadCount = 0;
  for (var i = 0; i < inbox.length; i++) { if (!inbox[i].read) unreadCount++; }

  if (unreadCount > 0) {
    drawText(ctx, unreadCount + ' UNREAD', 520, 12, PAL.emailUnread, 1);
  } else {
    drawText(ctx, 'ALL READ', 530, 12, PAL.positive, 1);
  }

  // Left panel: email list (40%)
  var listW = 240;
  ctx.fillStyle = PAL.emailBg;
  ctx.fillRect(0, 30, listW, 306);

  for (var i = 0; i < inbox.length; i++) {
    var email = inbox[i];
    var ey = 34 + i * 40;
    if (ey > 300) break;

    if (i === sel) {
      ctx.fillStyle = PAL.emailSelected;
      ctx.fillRect(0, ey, listW, 38);
    }

    // Unread dot
    if (!email.read) {
      ctx.fillStyle = PAL.emailNew;
      ctx.beginPath();
      ctx.arc(10, ey + 12, 3, 0, Math.PI * 2);
      ctx.fill();
    }

    drawText(ctx, email.from, 18, ey + 4, email.read ? PAL.emailRead : PAL.textPrimary, 1);
    var maxSubjW = listW - 24;
    var subj = email.subject;
    while (measureText(subj, 1).width > maxSubjW && subj.length > 3) subj = subj.slice(0, -1);
    if (subj.length < email.subject.length) subj += '..';
    drawText(ctx, subj, 18, ey + 16, email.read ? PAL.emailRead : PAL.textSecondary, 1);

    ctx.fillStyle = '#1a2a4e';
    ctx.fillRect(10, ey + 38, listW - 20, 1);
  }

  // Right panel: email detail (60%)
  ctx.fillStyle = '#111122';
  ctx.fillRect(listW, 30, 640 - listW, 306);

  if (sel >= 0 && sel < inbox.length) {
    var email = inbox[sel];
    var rx = listW + 16;
    var rightW = 640 - listW - 32;

    drawText(ctx, email.subject, rx, 44, PAL.textAccent, 2);
    drawText(ctx, 'FROM: ' + email.from, rx, 68, PAL.textSecondary, 1);

    ctx.fillStyle = '#222244';
    ctx.fillRect(rx, 82, rightW, 1);

    var bodyLines = wrapText(email.body, rightW, 1);
    for (var i = 0; i < bodyLines.length; i++) {
      drawText(ctx, bodyLines[i], rx, 92 + i * 14, PAL.textPrimary, 1);
    }

    // Draw choices if email has them and is not yet read
    if (email.choices && !email.read) {
      var choiceY = 92 + bodyLines.length * 14 + 16;
      for (var c = 0; c < email.choices.length; c++) {
        var isActive = (gs.uiState === 'email_choice' && c === gs.choiceIndex);
        ctx.fillStyle = isActive ? PAL.choiceActive : PAL.buttonBg;
        ctx.fillRect(rx, choiceY, rightW, 22);
        if (isActive) {
          ctx.strokeStyle = PAL.choiceBorder;
          ctx.lineWidth = 1;
          ctx.strokeRect(rx, choiceY, rightW, 22);
          drawText(ctx, '>', rx + 4, choiceY + 4, PAL.textAccent, 2);
        }
        drawText(ctx, email.choices[c].text, rx + 22, choiceY + 6, isActive ? PAL.textPrimary : PAL.textSecondary, 1);
        choiceY += 26;
      }
    } else if (email.read) {
      var readY = 92 + bodyLines.length * 14 + 16;
      drawText(ctx, '[READ]', rx, readY, PAL.emailRead, 1);
    }
  }

  // Bottom bar
  ctx.fillStyle = PAL.hudBg;
  ctx.fillRect(0, 336, 640, 24);

  if (gs.uiState === 'email_choice') {

function handleInput(e) {
  var key = e.code;
  if (['ArrowUp','ArrowDown','ArrowLeft','ArrowRight','Space','Tab','Enter','KeyE'].indexOf(key) !== -1) e.preventDefault();

  var gs = gameState;

  if (gs.gameOver && gs.uiState === 'game_over') {
    if (key === 'Enter' || key === 'Space') location.reload();
    return;
  }

  switch (gs.uiState) {
    case 'inbox': handleInboxInput(key); break;
    case 'email_choice': handleEmailChoiceInput(key); break;
    case 'action_menu': handleActionMenuInput(key); break;
    case 'action_result': handleActionResultInput(key); break;
    case 'evening_summary': handleEveningSummaryInput(key); break;
  }
}

function handleInboxInput(key) {
  var gs = gameState;
  var inbox = gs.inbox;
  if (key === 'ArrowUp' || key === 'KeyW') {
    gs.selectedIndex = Math.max(0, gs.selectedIndex - 1);
  } else if (key === 'ArrowDown' || key === 'KeyS') {
    gs.selectedIndex = Math.min(inbox.length - 1, gs.selectedIndex + 1);
  } else if (key === 'Enter' || key === 'Space') {
    if (gs.selectedIndex >= 0 && gs.selectedIndex < inbox.length) {
      var email = inbox[gs.selectedIndex];
      if (!email.read) {
        if (email.choices && email.choices.length > 0) {
          // Enter choice mode
          gs.uiState = 'email_choice';
          gs.choiceIndex = 0;
        } else {
          // No choices, just mark read
          markEmailRead(gs.selectedIndex);
        }
      }
    }
  } else if (key === 'KeyE') {
    if (allEmailsRead()) exitEmailPhase();
  }
}

function handleEmailChoiceInput(key) {
  var gs = gameState;
  var email = gs.inbox[gs.selectedIndex];
  if (!email || !email.choices) { gs.uiState = 'inbox'; return; }

  if (key === 'ArrowUp' || key === 'KeyW') {
    gs.choiceIndex = Math.max(0, gs.choiceIndex - 1);
  } else if (key === 'ArrowDown' || key === 'KeyS') {
    gs.choiceIndex = Math.min(email.choices.length - 1, gs.choiceIndex + 1);
  } else if (key === 'Enter' || key === 'Space') {
    selectEmailChoice(email, gs.choiceIndex);
    markEmailRead(gs.selectedIndex);
    gs.uiState = 'inbox';
  } else if (key === 'Escape' || key === 'KeyX') {
    gs.uiState = 'inbox';
  }
}

function handleActionMenuInput(key) {
  var gs = gameState;
  var actions = ENGINE_ACTIONS;
  if (key === 'ArrowUp' || key === 'KeyW') {
    gs.selectedIndex = Math.max(0, gs.selectedIndex - 1);
  } else if (key === 'ArrowDown' || key === 'KeyS') {
    gs.selectedIndex = Math.min(actions.length - 1, gs.selectedIndex + 1);
  } else if (key === 'Enter' || key === 'Space') {
    var action = actions[gs.selectedIndex];
    if (!action) return;
    var cost = AP.costs[action.apKey] || 0;
    var affordable = cost === 0 || AP.current >= cost;
    var enabled = action.enabled(gs) && affordable;
    if (!enabled) return;

    var result = performAction(action.id);
    if (result.success) {
      gs.actionResult = result;
      gs.uiState = 'action_result';
    }
  }
}

function handleActionResultInput(key) {
  var gs = gameState;
  if (key === 'Space' || key === 'Enter') {
    gs.actionResult = null;
    if (AP.current <= 0) {
      startEvening();
    } else {
      gs.uiState = 'action_menu';
      gs.selectedIndex = 0;
    }
  }
}

function handleEveningSummaryInput(key) {
  var gs = gameState;
  if (key === 'Enter' || key === 'Space') {
    if (gs.gameOver) {
      gs.uiState = 'game_over';
    } else {
      advanceDay();
    }
  }
}

window.addEventListener('keydown', handleInput);

}
